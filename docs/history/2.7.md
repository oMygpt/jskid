// ==UserScript==
// @name         AMAC 培训系统 - 究极光速挂机助手
// @namespace    http://tampermonkey.net/
// @version      2.6
// @description  修复视频重播问题。隐身模式：取消跳帧，最后30秒原速结算，彻底杜绝循环播放。
// @author       Gemini CLI
// @match        *://peixun.amac.org.cn/*
// @grant        none
// @run-at       document-start
// ==/UserScript==

(function() {
    'use strict';

    console.log('--- AMAC GOD MODE v2.6 [FIX REPLAY] ACTIVATED ---');

    // 1. 拦截干扰事件
    function blockEvents() {
        var noop = function(e) { e.stopImmediatePropagation(); return false; };
        window.addEventListener('blur', noop, true);
        window.addEventListener('mouseleave', noop, true);
        window.addEventListener('focusout', noop, true);
        try {
            Object.defineProperty(document, 'visibilityState', { get: function() { return 'visible'; }, configurable: true });
            Object.defineProperty(document, 'hidden', { get: function() { return false; }, configurable: true });
        } catch(e) {}
    }

    // 2. 视频核心注入
    function hackVideo(v) {
        if (!v || v._hacked) return;
        v._hacked = true;
        v.muted = true;
        v.loop = false; // 强制禁用循环播放

        // 拦截暂停
        v.pause = (function(orig) {
            return function() {
                if (v.ended || v._hasFinished) return orig.apply(this, arguments);
                console.log('拦截自动暂停');
            };
        })(v.pause);

        // 监听结束事件
        v.addEventListener('ended', function() {
            v._hasFinished = true;
            console.log('视频正常播放结束，已锁定状态防止重播。');
        });

        // 核心监控循环
        var monitor = setInterval(function() {
            // 判定是否已经完成（兼容性判定）
            if (v.ended || (v.duration > 0 && v.currentTime / v.duration > 0.999)) {
                v._hasFinished = true;
            }

            // 如果已经完成，彻底停止干预，并销毁监控
            if (v._hasFinished) {
                v.playbackRate = 1.0;
                clearInterval(monitor);
                return;
            }

            // 强制播放 (仅在未完成时)
            if (v.paused) v.play().catch(function(){});

            // 动态调速策略：
            var remain = v.duration - v.currentTime;
            var progress = v.currentTime / v.duration;

            if (remain < 30 || progress > 0.96) {
                if (v.playbackRate !== 1.0) {
                    v.playbackRate = 1.0;
                    console.log('>>> 进入结算区：1.0x 运行...');
                }
            } else {
                if (v.playbackRate !== 12.0) {
                    v.playbackRate = 12.0;
                }
            }
        }, 1000);
    }

    // 3. 递归探测视频
    function isAnyVideoRunning(root) {
        var videos = Array.from(root.querySelectorAll('video'));
        var running = videos.some(function(v) { return !v.ended && !v._hasFinished; });
        if (running) return true;

        var iframes = Array.from(root.querySelectorAll('iframe'));
        for (var i = 0; i < iframes.length; i++) {
            try {
                var doc = iframes[i].contentDocument || iframes[i].contentWindow.document;
                if (doc && isAnyVideoRunning(doc)) return true;
            } catch(e) {}
        }
        return false;
    }

    // 4. 智能点击
    function autoClick() {
        var confirmKeywords = ['确定', '确认', '继续', '好的', '阅读', '同意', '知道了', '我知道了'];
        var jumpKeywords = ['下一节', '进入测验', '开始练习', '去评价', '提交', '完成', '结束'];

        var running = isAnyVideoRunning(document);

        var elements = document.querySelectorAll('button, a, .btn, .ui-button, [role="button"], .layui-layer-btn0');
        elements.forEach(function(el) {
            var text = (el.innerText || "").replace(/\s+/g, '');
            if (!text || el.offsetParent === null) return;

            // 确认类：直接点
            if (confirmKeywords.some(function(k) { return text.indexOf(k) !== -1; })) {
                var now = Date.now();
                if (!el._lastClicked || now - el._lastClicked > 5000) {
                    el._lastClicked = now;
                    console.log('自动清理弹窗:', text);
                    el.click();
                }
                return;
            }

            // 跳转类：没视频跑才准点
            if (jumpKeywords.some(function(k) { return text.indexOf(k) !== -1; })) {
                if (running) return;

                if (!el._waiting) {
                    el._waiting = true;
                    console.log('视频已播完，等待 15 秒数据同步后跳转...');
                    setTimeout(function() {
                        el.click();
                        el._waiting = false;
                    }, 15000);
                }
            }
        });
    }

    // 5. 扫描引擎
    function scan(root) {
        try {
            root.querySelectorAll('video').forEach(hackVideo);
            root.querySelectorAll('iframe').forEach(function(f) {
                try {
                    var doc = f.contentDocument || f.contentWindow.document;
                    if (doc) scan(doc);
                } catch(e) {}
            });
        } catch(e) {}
    }

    // 启动
    blockEvents();
    setInterval(function() {
        scan(document);
        autoClick();
    }, 2500);

})();
